main:
cpi	r30,	255
breq	init
retInit:

cp	r30,	r20
breq 	offRed
retRed:

cp	r30,	r21
breq	offGreen
retGreen:

cp	r30,	r22
breq	offBlue
retBlue:

inc	r30
rjmp main



init:
cbi	PORTB,	0	;2	turn on Red
cbi	PORTB,	3	;2	turn on Green
cbi	PORTB,	1	;2	turn on Blue

ldi	r30,	0	;1	reinit counter
rjmp	retInit

offRed:
sbi	PORTB,	0	;2	turn off Red
rjmp	retRed

offGreen:
sbi	PORTB,	3	;2	turn off Green
rjmp	retGreen

offBlue:
sbi	PORTB,	1	;2	turn off Blue
rjmp	retBlue
